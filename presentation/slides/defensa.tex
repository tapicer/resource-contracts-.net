\section{Introducción}

\begin{frame}{Verificación}
\begin{itemize}
	\spitemsma{¿Por qué analizar el consumo de memoria?}
		\begin{itemize}
			\spitemsma{Sistemas con requerimientos estrictos de memoria: tiempo real, embebidos, misión crítica}
			\spitemsma{Entornos donde se cobra el uso de recursos (cloud)}
			\spitemmed{Es valioso contar con un \textit{certificado} de la memoria requerida para una ejecución segura}
		\end{itemize}
	\pause
	\spitemsma{Inferir vs. Verificar}
		\begin{itemize}
			\spitemsma{Análisis de consumo de memoria en general: es indecidible}
			\spitemsma{Verificar es ``más fácil'' que inferir, analogía del laberinto}
			\spitemsma{El usuario \textit{razona}, tiene un conocimiento profundo del programa}
			\spitemmed{La combinación usuario + verificación puede ser mejor que la inferencia}
		\end{itemize}
	\pause
	\spitemsma{Verificación de consumo de memoria: área poco explorada}
\end{itemize}
\notesp{Sistemas donde se cobra: no queremos pagar de más, pero hay que saber cuánto consume}
\notesp{Contar diferencias entre inferir y verificar}
\notesp{Queremos conocer la memoria necesaria para que el programa se ejecute de forma segura}
\notesp{Laberinto: difícil de resolver, fácil de verificar solución}
\notesp{Podemos pensar que tenemos un problema (saber el consumo, como el laberinto) y usuario nos da la solución, tenemos que verificarla}
\notesp{Existen para inferir: Diego, Martín, extensión de Gastón y Matías}
\notesp{Existen para verificar, implementaciones: JML, Spec\#, Code Contracts}
\end{frame}

{
  Contract.Requires(n >= 0);

\section{Anotaciones}

\begin{frame}{Modelando el uso de memoria}
\begin{itemize}
	\spitemsma{Los lenguajes orientados a objetos modernos usan un GC para administrar la memoria, impredecible}
	\pause
	\spitemsma{Vamos a usar un modelo del consumo de memoria predecible}
	\pause
		\begin{itemize}
			\spitemsma{Sobreaproxima los requerimientos reales, permitiendo obtener una cota del consumo}
			\vspace{5pt}
			\pause
			\spitemsma{Asumimos que los objetos son recolectados al final del método}
			\vspace{5pt}
			\pause
			\spitemsma{Sólo tiene en cuenta objetos \textbf{creados} por el método analizado y los métodos que este invoca}
			\pause
			\begin{itemize}
				\spitemsma{Objetos temporales: no necesarios después de la ejecución del método, se pueden eliminar de memoria}
				\vspace{5pt}
				\spitemsma{Objetos residuales: exceden el tiempo de vida del método, deben seguir vivos luego de que finalice la ejecución del método}
			\end{itemize}
		\end{itemize}
\end{itemize}
\notesp{Necesitamos modelar el uso de memoria de una forma donde la alocación y liberación sea predecible}
\notesp{Obtenemos una cota superior: los GC son mejores (más agresivos) que al final del método, por eso sobreaproximamos}
\notesp{Partimos al conjunto de objetos creados por un método en dos: tmp y rsd}
\notesp{EXPLICAR RÁPIDO Y DETALLAR EN EL EJEMPLO}
\notesp{Temporales: objetos usados para un cálculo o proceso auxiliar durante la ejecución}
\notesp{Residuales: de diferentes formas (devuelto, parámetros, globales)}
\end{frame}

\begin{frame}{Verificación de anotaciones de tiempo de vida}
\begin{itemize}
	\spitemsma{Necesitamos verificar que las anotaciones \smono{Contract.Memory.DestTmp}, \smono{Contract.Memory.DestRsd}, \smono{Contract.Memory.AddTmp} y \smono{Contract.Memory.AddRsd} son correctas}
	\pause
	\spitemsma{Construimos un Points-to Graph (PTG) del método haciendo un análisis de points-to, grafo dirigido donde los nodos son objetos (o conjuntos de) y los arcos referencias entre ellos}
	\pause
	\spitemsma{A partir del PTG hacemos un análisis de escape de objetos para ver qué objetos exceden el tiempo de vida del método}
	\pause
	\spitemsma{Verificamos}
	\begin{itemize}
		\spitemsma{Que los objetos escapen del método si y sólo si están anotados como residuales}
		\spitemsma{Que los objetos residuales escapen del método a través de la expresión indicada}
	\end{itemize}
\end{itemize}
\notesp{Para que los contratos sean correctos, las anotaciones de tiempo de vida tienen que serlo}
\end{frame}

\begin{frame}{Verificación con aritmética no lineal}
\begin{itemize}
	\spitemmed{El verificador de Code Contracts sólo soporta aritmética lineal}
	\pause

	\pause
	\spitemmed{Durante la verificación}
	\begin{itemize}
		\spitemmed{Usamos la herramienta para calcular expresiones polinomiales e incrementamos los contadores con valores pre-calculados}
		\spitemmed{Obtenemos aserciones acerca de la validez de los contratos y las brindamos al verificador mediante la anotación \smono{Assume}}
	\end{itemize}
	\pause
	\spitemmed{Requerimos para estos casos que se anote un \smono{IterationSpace} para los ciclos}

\section{Limitaciones y trabajo futuro}

\begin{frame}{Limitaciones}
\begin{itemize}
	\item{Estamos limitados por las capacidades de:}
		\begin{itemize}
			\vspace{3pt}
			\item El verificador estático
			\vspace{3pt}
			\item La herramienta para resolver operaciones con aritmética no lineal
			\vspace{3pt}
			\pause
			\item{$\leadsto$ El diseño de las técnicas permite reemplazarlas por herramientas con mayores o mejores capacidades cuando existan}
		\end{itemize}
	\vspace{8pt}
	\pause
	\item{El análisis de tiempo de vida es sobre-aproximado}
	\pause
	\begin{itemize}
		\vspace{3pt}
		\item{$\leadsto$ El análisis de points-to es un problema indecidible}
		\vspace{3pt}
		\item{$\leadsto$ Permitimos obviar la verificación para casos en que falla}
	\end{itemize}
	\vspace{8pt}
	\pause
	\item{Requerimos que el usuario provea todas las anotaciones de tiempo de vida para la verificación, y los espacios de iteración para utilizar la herramienta para aritmética no lineal}
	\pause
	\begin{itemize}
		\vspace{3pt}
		\item{$\leadsto$ En un futuro pensamos inferir estas anotaciones}
	\end{itemize}
\end{itemize}
\notesp{Limitaciones, tanto por las herramientas usadas como inherentes a las técnicas propuestas}
\notesp{Como dependemos de otras herramientas (verificador, aritmética), siempre vamos a estar limitados}
\notesp{Análisis de tiempo de vida, siempre va a ser un problema}
\notesp{Muchas anotaciones requeridas, inferencia ayudaría}
\end{frame}

\begin{frame}{Trabajo futuro}
\begin{itemize}
	\item{Incorporar capacidades de inferencia permitiendo al usuario proveer sólo los contratos de consumo y no las anotaciones adicionales actualmente requeridas}
	\begin{itemize}
		\item{De las anotaciones de tiempo de vida}
		\item{De los espacios de iteración de los ciclos}
	\end{itemize}
	\pause
	\item{Experimentar con otros verificadores estáticos}
	\begin{itemize}
		\item{\mono{ESC/Java2} para Java: requeriría implementar la instrumentación para Java}
		\item{\mono{Z3} para Java o .NET: requeriría traducir el código a un lenguaje intermedio que utiliza (\mono{Boogie})}
	\end{itemize}
	\pause
	\item{Extender las capacidades del análisis con aritmética no lineal}
	\begin{itemize}
		\item{Mejorar la capacidad de cálculo de máximos entre polinomios}
		\item{Evaluar/modificar la herramienta utilizada o reemplazarla por otra}
	\end{itemize}
	\pause
	\item{Mejorar la usabilidad e integración con la IDE}
	\begin{itemize}
		\item Desarrollando un plug-in que autocomplete anotaciones de acuerdo a los contratos existentes
	\end{itemize}
\end{itemize}
\notesp{Podemos integrar otros trabajos sobre inferencia}
\notesp{Hay muchos verificadores, Z3 podría integrarse sin ningún trabajo cuando Code Contracts lo soporte}
\notesp{Ya desarrollamos un plugin para instalar el tool como ext de VS y habilitar/deshabilitar contratos de mem, la idea es extenderlo para asistencia de anotaciones}
